介绍几种比较特殊的函数定义以及使用

带有可变参数的函数变现形式

函数可以被嵌套

函数可以作为另一个函数的返回值。

函数也可以当做参数传入另一个函数。

func hasAnyMatches(list: [Int], codition:(Int)-> Bool) -> Bool {

    for item in list {
        if codition(item) {
            return true
        }
    }
    return false
}

func lessThanTen(number: Int) -> Bool {
    return number < 10
}

var numbers = [12, 23, 9, 50]
hasAnyMatches(list: numbers, codition: lessThanTen)
